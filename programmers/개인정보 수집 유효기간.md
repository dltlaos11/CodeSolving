```javascript
function solution(today, terms, privacies) {
  var answer = [];

  let terms_arr = terms.map((e) => e.split(" "));

  let term_obj = {};
  let termObj = terms_arr.map((e) => {
    term_obj[e[0]] = e[1];
  }); // {A: '6', B: '12', C: '3'}

  let termKeys = Object.keys(term_obj); // ['A', 'B', 'C']

  let pri = privacies.map((e) => e.split(" "));

  for (let i = 0; i < pri.length; i++) {
    let year = 0;
    let month = 0;
    let date = 0;
    let today_date = "";

    if (termKeys.includes(pri[i][1])) {
      month += Number(pri[i][0].substring(5, 7)) + Number(term_obj[pri[i][1]]);
      if (month > 12) {
        year += Number(pri[i][0].substring(0, 4)) + 1;
        month -= 12;
        if (month === 0) {
          month = 1;
        }
      } else if (month <= 12) {
        year += Number(pri[i][0].substring(0, 4));
      }
      date += Number(pri[i][0].substring(8, 10)) - 1;
      if (date === 0) {
        date = 28;
        month -= 1;
      }
      if (month < 10) {
        month = month.toString();
        month = "0" + month;
      }
      if (date < 10) {
        date = date.toString();
        date = "0" + date;
      }

      const val = year.toString() + month.toString() + date.toString();
      today_date +=
        today.substring(0, 4) + today.substring(5, 7) + today.substring(8, 10);
      console.log(Number(val), Number(today_date));

      if (Number(val) < Number(today_date)) {
        answer.push(i + 1);
      }
    }
  }
  return answer;
}
```

## 다른 사람 풀이

```javascript
function solution(today, terms, privacies) {
  var answer = [];
  var [year, month, date] = today.split(".").map(Number);
  var todates = year * 12 * 28 + month * 28 + date;
  var t = {};
  terms.forEach((e) => {
    let [a, b] = e.split(" ");
    t[a] = Number(b);
  });
  privacies.forEach((e, i) => {
    var [day, term] = e.split(" ");
    day = day.split(".").map(Number);
    var dates = day[0] * 12 * 28 + day[1] * 28 + day[2] + t[term] * 28;
    if (dates <= todates) answer.push(i + 1);
  });
  return answer;
}
```

```javascript
let term_obj = {};
let termObj = terms_arr.map((e) => {
  term_obj[e[0]] = e[1];

// 남의 방식
var t = {};
terms.forEach((e) => {

```
