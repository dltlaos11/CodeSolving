import sys

input = sys.stdin.readline
rows, columns = map(int, input().split())

answer =[]

matrix = [[0 for j in range(columns+1)] for i in range(rows+1)]
num = 1
for i in range(1, rows+1):
    for j in range(1, columns+1):
        matrix[i][j] = ((i-1) * columns + j)
# matrix = [[0 for j in range(columns)] for i in range(rows)]
# for i in range(0, rows):
#     for j in range(0, columns):
#         matrix[i][j] = i * rows + j + 1
# 정방행렬이 아닌 경우 [[1, 2, 3, 4], [7, 8, 9, 10], [13, 14, 15, 16], [19, 20, 21, 22], [25, 26, 27, 28], [31, 32, 33, 34]] 문제..
# TC 3부터 문제
print(matrix)

queries =[[1,1,100,97]]
for x1, y1, x2, y2 in queries:
    init = matrix[x1][y1]
    Min = init # queries내 계속 바뀌는

    #x1 좌측 1 4
    for i in range(x1, x2):
        temp = matrix[i + 1][y1]
        matrix[i][y1] = temp
        Min = min(Min, temp)
    # y2 좌측
    for i in range(y1, y2):
        temp = matrix[x2][i + 1]
        matrix[x2][i] = temp
        Min = min(Min, temp)
    # x1 우측
    for i in range(x2, x1, -1):
        temp = matrix[i - 1][y2]
        matrix[i][y2] = temp
        Min = min(Min, temp)
    # y1 우측
    for i in range(y2, y1, -1):
        temp = matrix[x1][i - 1]
        matrix[x1][i] = temp
        Min = min(Min, temp)

    matrix[x1][y1 + 1] = init
    answer.append(Min)

print(answer)
