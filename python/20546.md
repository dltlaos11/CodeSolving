import sys
input = sys.stdin.readline

stock = int(input())
numbers = list(map(int, input().split()))
def BNP(nums):
    cnt = 0
    for i in nums:
        if i <= stock:
            if cnt > 0: continue
            cnt = stock / i
    return int(cnt) * nums[-1]


def TIMING(nums, stock):
    cnt = 0
    stk = stock
    for i in range(len(nums)):

        if i < len(nums) - 3:
            if nums[i] < nums[i + 1] < nums[i + 2] < nums[i + 3]:
                if (int(cnt) * nums[i + 3]) == 0:
                    continue
                else:
                    return int(cnt) * nums[i + 3]
            if nums[i] > nums[i + 1] > nums[i + 2] > nums[i + 3]:
                cnt += stk / nums[i + 3]
                stk = stk % nums[i + 3]
    return int(cnt) * nums[-1]

if BNP(numbers) > TIMING(numbers, stock):
    print("BNP")
elif BNP(numbers) < TIMING(numbers, stock):
    print("TIMING")
else:
    print("SAMESAME")
